```Python
class Stack:
     def __init__(self):
        self.items = []

    def isEmpty(self):
         return self.items == []

    def push(self, item):
         self.items.append(item)

    def pop(self):
         return self.items.pop()

    def peek(self):
         return self.items[len(self.items)-1]

    def size(self):
         return len(self.items)
```

```Python
# 금속 이온이 2가 3가 일때도 적용시켜주기
def gcd(a, b):
    while (b != 0):
        temp = a % b
        a = b
        b = temp
    return abs(a)
#최대공약수 함수 정의
```
