    def __init__(self,k=[]):

    def push(self, item):
        self.items.append(item)
    def pop(self):
        return self.items.pop()
    def isEmpty(self):
        return len(self.items) == 0
    def peek(self):
        return self.items[-1]
    def size(self):
        return len(self.items)
